numpy==1.26.4

# headless, not the full GUI build: this app only ever calls cv2's pure
# array/image-processing functions (cvtColor, inRange, findContours...),
# never cv2.imshow/namedWindow -- the full opencv-python package needs
# system GUI libraries (libGL, libglib2.0, etc.) that cloud containers
# don't ship by default and that have been unreliable to apt-get install
# on Streamlit Cloud (broken dependency chains on their current base
# image). Headless sidesteps that whole problem class since it doesn't
# link against those libraries at all, and needs zero packages.txt entries.
#
# LOCAL USE ONLY: calibrate_camera.py and trigger_system.py's continuous
# local-webcam loop DO call cv2.imshow/cv2.waitKey and need the real GUI
# build. If you want to run those locally: `pip uninstall opencv-python-
# headless && pip install opencv-python` in your own venv (don't do this
# in the deployed app's requirements.txt, or Streamlit Cloud breaks again).
opencv-python-headless==4.9.0.80
streamlit==1.32.2
matplotlib==3.8.4
requests==2.31.0
